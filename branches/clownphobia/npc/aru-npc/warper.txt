-	script	Warpmaster::warpra	721,{
	set @image,rand(3);
	if (@image==0) cutin "kafra_07",2;
	if (@image==1) cutin "kkafra_07",2;
	if (@image==2) cutin "ukafra_07",2;
	mes "[Warpmaster]";
	mes "Hello,";
	mes "I can warp you to any Town and Dungeon!";
	mes "Were do you want to go?";
	callfunc "warpra_menu";
	close;
}
