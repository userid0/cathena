// Auto Start Agit NPC
prontera.gat,116,72,0	script	Agit_Event	-1,{
OnInit:
	if ($AgitStarted == 1) goto AgitReboot;
	Break;
OnClock1900:
	Announce "The War Of Emperium has begun!",8;
	AgitStart;
	set $AgitStarted,1;
	Break;
OnClock2100:
	Announce "The War Of Emperium is over!",8;
	AgitEnd;
	set $AgitStarted,0;
	Break;
AgitReboot:
	AgitStart;
	set $AgitStarted,1;
	Break;
}
