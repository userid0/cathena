		mes "[Airport Staff]";

		mes "is 1,200 zeny, but if you've";
		mes "got a Free Ticket for Airship,"; 
		mes "the fee will be waived. Will";

			warp "airport.gat",148,51;
			close;
	L_Cancel:

		mes "Thank you and";
		mes "have a nice day.";
		close;
