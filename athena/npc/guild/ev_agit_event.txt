// Auto Start Agit NPC
- script Agit_Event -1,{
	end;

OnAgitInit:
	if(gettime(3) >= 21 || gettime(3) < 11) goto L_Stop;

OnClock1100:
//	if(gettime(4)!=1 || gettime(4)!=3 || gettime(4)!=6) end;
	Announce "The War Of Emperium has begun!",8;
	AgitStart;
	Break;

OnClock2100:
L_Stop:
//	if(gettime(4)!=1 || gettime(4)!=3 || gettime(4)!=6) end;
	Announce "The War Of Emperium is over!",8;
	AgitEnd;
	Break;
}
