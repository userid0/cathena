	mes "[Reset Girl]";
	mes "I am a Reset Girl.";
	mes "Reset Stats: 100,000z";
	mes "Reset Skills: 100,000z";
	mes "Reset Both: 200,000z";
	next;
	mes "Please select the service you want:";
	menu "^FF3355Reset Skills",L0,"Reset Stats",L1,"Reset Both^000000",L2,"Quit",LEnd;
	
	L0:
		mes "[Reset Girl]";
		if (Zeny < 100000) goto NeedZenys;
		mes "Alright, here we go now.. Remember, changes won't take effect until you log back on!";
		set Zeny,zeny-100000;
		ResetSkill;
		goto WellDone;
	L1:
		mes "[Reset Girl]";
		if (Zeny < 100000) goto NeedZenys;
		mes "Alright, here we go now.. Remember, changes won't take effect until you log back on!";
		set Zeny,zeny-100000;
		ResetStatus;
		goto WellDone;
	L2:
		mes "[Reset Girl]";
		if (Zeny < 200000) goto NeedZenys;
		mes "Alright, here we go now.. Remember, changes won't take effect until you log back on!";
		set Zeny,zeny-200000;
		ResetSkill;
		ResetStatus;
		goto WellDone;

	WellDone:
		mes "[Reset Girl]";
		mes "You have Successfuly Reset your Skills or Stats";
		mes "Please Relogin To Make sure the Changes take Effect ASAP.";
		mes "Use em Wisely.";
		mes "Good luck in the Future. Bye~";
		close;

		NeedZenys:
			mes "Sorry, you don't have enough Zeny.";
			mes "Come back when you get more Zeny. Im always here.";
			close;
	LEnd:
		mes "[Reset Girl]";
		mes "Come back soon";
		close;	
}
