prontera.gat,150,184,5	script	Dog	81,{	
	mes "[Dog]";
	mes "Woof Woof Woof Woof!!";
	next;
	menu "Heal",L_HEAL,"Quit",CANCEL;
	
	L_HEAL:
		mes "[Dog]";
		mes "Woof (Need a Heal)?";
		next;
		menu "Yes",-,"No",L_HEAL_LV100_N;
		
		if(Zeny>=0) goto L_HEALING;
	L_HEALING:
		set Zeny,zeny-0;
		heal 30000,30000;
		mes "[Dog]";
		mes "Woof! Woof woof! (Healed! Have fun!)";
		close;
			
			L_HEAL_LV100_F:
			mes "[Dog]";
			mes "Woof";
			close;
			
			L_HEAL_LV100_N:
			mes "[Dog]";
			mes "Woof Woof (Bye Bye!)";
			close;
					
	CANCEL:
	mes "[Dog]";
	mes "Woof Woof! (Bye Bye!)";
	close;
}
