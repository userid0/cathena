pay_gld.gat,301,118,1	script	Tiantan Knight	733,{

	set @GIDf2,GetCastleData("payg_cas02.gat",1);
	if (@GIDf2 == 0) goto Lclear;
	if (getcharid(2) == @GIDf2) goto LStartg;
	mes "[Tiantan Knight]";
	mes "You're not allies of my King! Halt your Pursuit now or face death beyond your worst fear.";
	goto L_EXIT;

Lclear:
	mes "[Tiantan Knight]";
	mes "When the time is ripe, the Uncrowned shall be Crowned once again... and the broken blade shall be one again...";
	close;

L_EXIT:
	close;

LStartg:
	mes "[Tiantan Knight]";
	mes "Welcome. ^ff0000" + GetGuildName(@GIDf2) + "^000000 members";
	mes "May the light be with you...";
	next;
	menu "Save",Lsave,"Protect Emperium",protect_emp,"Leave Here",leave,"Cancel",Lcancel;

	Lsave:
		savepoint "pay_gld.gat",298,117;
		mes "[Tiantan Knight]";
		mes "Your respawn point has saved. Thank you.";
		close;

	protect_emp:
		warp "payg_cas02.gat",	38,	24;	close;

	leave:
		warp "prontera.gat",	154,	194;	close;

	Lcancel:
		close;

}
