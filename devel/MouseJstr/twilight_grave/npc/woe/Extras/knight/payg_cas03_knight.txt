pay_gld.gat,313,296,5	script	Fuying Knight	733,{

	set @GIDf3,GetCastleData("payg_cas03.gat",1);
	if (@GIDf3 == 0) goto Lclear;
	if (getcharid(2) == @GIDf3) goto LStartg;
	mes "[Fuying Knight]";
	mes "You're not allies of my King! Halt your Pursuit now or face death beyond your worst fear.";
	goto L_EXIT;

Lclear:
	mes "[Fuying Knight]";
	mes "When the time is ripe, the Uncrowned shall be Crowned once again... and the broken blade shall be one again...";
	close;

L_EXIT:
	close;

LStartg:
	mes "[Fuying Knight]";
	mes "Welcome. ^ff0000" + GetGuildName(@GIDf3) + "^000000 members";
	mes "May the light be with you...";
	next;
	menu "Save",Lsave,"Protect Emperium",protect_emp,"Leave Here",leave,"Cancel",Lcancel;

	Lsave:
		savepoint "pay_gld.gat",315,292;
		mes "[Fuying Knight]";
		mes "Your respawn point has saved. Thank you.";
		close;

	protect_emp:
		warp "payg_cas03.gat",	268,	264;	close;

	leave:
		warp "prontera.gat",	154,	194;	close;

	Lcancel:
		close;

}
