//WoE Castle Knight NPC Script by holyAngelX

prt_gld.gat,192,241,5	script	Guindull Knight	733,{

	set @GIDp5,GetCastleData("prtg_cas05.gat",1);
	if (@GIDp5 == 0) goto Lclear;
	if (getcharid(2) == @GIDp5) goto LStartg;
	mes "[Guindull Knight]";
	mes "You're not allies of my King! Halt your Pursuit now or face death beyond your worst fear.";
	goto L_EXIT;

Lclear:
	mes "[Guindull Knight]";
	mes "When the time is ripe, the Uncrowned shall be Crowned once again... and the broken blade shall be one again...";
	close;

L_EXIT:
	close;

LStartg:
	mes "[Guindull Knight]";
	mes "Welcome. ^ff0000" + GetGuildName(@GIDp5) + "^000000 members";
	mes "May the light be with you...";
	next;
	menu "Save",Lsave,"Protect Emperium",protect_emp,"Leave Here",leave,"Cancel",Lcancel;

	Lsave:
		savepoint "prt_gld.gat",193,235;
		mes "[Guindull Knight]";
		mes "Your respawn point has saved. Thank you.";
		close;

	protect_emp:
		warp "prtg_cas05.gat",	265,	265;	close;

	leave:
		warp "prontera.gat",	154,	194;	close;

	Lcancel:
		close;

}
