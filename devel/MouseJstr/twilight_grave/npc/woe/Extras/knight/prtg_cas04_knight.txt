prt_gld.gat,124,233,5	script	Squagul Knight	733,{

	set @GIDp4,GetCastleData("prtg_cas04.gat",1);
	if (@GIDp4 == 0) goto Lclear;
	if (getcharid(2) == @GIDp4) goto LStartg;
	mes "[Squagul Knight]";
	mes "You're not allies of my King! Halt your Pursuit now or face death beyond your worst fear.";
	goto L_EXIT;

Lclear:
	mes "[Squagul Knight]";
	mes "When the time is ripe, the Uncrowned shall be Crowned once again... and the broken blade shall be one again...";
	close;

L_EXIT:
	close;

LStartg:
	mes "[Squagul Knight]";
	mes "Welcome. ^ff0000" + GetGuildName(@GIDp4) + "^000000 members";
	mes "May the light be with you...";
	next;
	menu "Save",Lsave,"Protect Emperium",protect_emp,"Leave Here",leave,"Cancel",Lcancel;

	Lsave:
		savepoint "prt_gld.gat",126,237;
		mes "[Squagul Knight]";
		mes "Your respawn point has saved. Thank you.";
		close;

	protect_emp:
		warp "prtg_cas04.gat",	291,	13;	close;

	leave:
		warp "prontera.gat",	154,	194;	close;

	Lcancel:
		close;

}
