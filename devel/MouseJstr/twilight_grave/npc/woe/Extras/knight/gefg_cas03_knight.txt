gef_fild13.gat,83,177,4	script	Isinlife Knight	733,{

	set @GIDg3,GetCastleData("gefg_cas03.gat",1);
	if (@GIDg3 == 0) goto Lclear;
	if (getcharid(2) == @GIDg3) goto LStartg;
	mes "[Isinlife Knight]";
	mes "You're not allies of my King! Halt your Pursuit now or face death beyond your worst fear.";
	goto L_EXIT;

Lclear:
	mes "[Isinlife Knight]";
	mes "When the time is ripe, the Uncrowned shall be Crowned once again... and the broken blade shall be one again...";
	close;

L_EXIT:
	close;

LStartg:
	mes "[Isinlife Knight]";
	mes "Welcome. ^ff0000" + GetGuildName(@GIDg3) + "^000000 members";
	mes "May the light be with you...";
	next;
	menu "Save",Lsave,"Protect Emperium",protect_emp,"Leave Here",leave,"Cancel",Lcancel;

	Lsave:
		savepoint "gef_fild13.gat",83,173;
		mes "[Isinlife Knight]";
		mes "Your respawn point has saved. Thank you.";
		close;

	protect_emp:
		warp "gefg_cas03.gat",	244,	166;	close;

	leave:
		warp "prontera.gat",	154,	194;	close;

	Lcancel:
		close;

}
